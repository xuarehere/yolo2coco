         ]
  # 被丢弃的信息，储存地址。如：图片中的所有的label都不在合并的类型中，因此丢弃       
  save_dir: "./tmp/"             
OLD:

                'BigTruck', 
                'Bus', 
                'Car', 
                'CoachesCar',
